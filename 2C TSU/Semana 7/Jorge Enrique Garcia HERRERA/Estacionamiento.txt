import java.util.Scanner;

/**
 *
 * @author jegh1
 */
public class Estacionamiento {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
    
       Scanner scanner = new Scanner(System.in);

        // 1. Crear el array de 10 espacios
        String[] espacios = new String[10];

        // Inicializar todos como LIBRE
        for (int i = 0; i < espacios.length; i++) 
            espacios[i] = "LIBRE";
        

        int opcion = 0;

        // 2. Menú interactivo
        while (opcion != 5) {

            System.out.println("\n--- SISTEMA DE ESTACIONAMIENTO ---");
            System.out.println("1. Ver estacionamiento");
            System.out.println("2. Estacionar auto");
            System.out.println("3. Retirar auto");
            System.out.println("4. Contar espacios ocupados");
            System.out.println("5. Salir");
            System.out.print("Selecciona una opción: ");

            opcion = scanner.nextInt();

            switch (opcion) {

                case 1:
                    // 3. Ver estacionamiento
                    for (int i = 0; i < espacios.length; i++) {
                        System.out.println("Espacio " + (i + 1) + ": " + espacios[i]);
                    }
                    break;

                case 2:
                    // 4. Estacionar auto
                    System.out.print("Ingresa número de espacio (1-10): ");
                    int lugar = scanner.nextInt() - 1;

                    if (lugar >= 0 && lugar < 10) {
                        if (espacios[lugar].equals("LIBRE")) {
                            espacios[lugar] = "OCUPADO";
                            System.out.println("Auto estacionado correctamente.");
                        } else {
                            System.out.println("Error: El espacio ya está ocupado.");
                        }
                    } else {
                        System.out.println("Número inválido.");
                    }
                    break;

                case 3:
                    // 5. Retirar auto
                    System.out.print("Ingresa número de espacio (1-10): ");
                    int retirar = scanner.nextInt() - 1;

                    if (retirar >= 0 && retirar < 10) {
                        if (espacios[retirar].equals("OCUPADO")) {
                            espacios[retirar] = "LIBRE";
                            System.out.println("Auto retirado correctamente.");
                        } else {
                            System.out.println("Error: El espacio ya está libre.");
                        }
                    } else {
                        System.out.println("Número inválido.");
                    }
                    break;

                case 4:
                    // 6. Contar espacios ocupados
                    int contador = 0;

                    for (String estado : espacios) {
                        if (estado.equals("OCUPADO")) {
                            contador++;
                        }
                    }

                    System.out.println("Espacios ocupados: " + contador);
                    break;

                case 5:
                    // 7. Salir
                    System.out.println("Saliendo del sistema...");
                    break;

                default:
                    System.out.println("Opción inválida.");
            }
        }

        scanner.close();
    }
}
