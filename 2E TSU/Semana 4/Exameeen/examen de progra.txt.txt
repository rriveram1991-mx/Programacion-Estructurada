 public static void main(String[] args) 
    {
        Scanner sc = new Scanner(System.in);
        int opcion = 0;
        double totalVentas = 0;
        int contadorVentas = 0;

        while (opcion != 3) //  ERROR 1: Punto y coma hace que el while quede vacío (bucle infinito)
        {   

            System.out.println("1. Registrar venta");
            System.out.println("2. Mostrar total vendido");
            System.out.println("3. Salir");
            System.out.print("Seleccione una opcion: ");
            opcion = sc.nextInt();

            switch (opcion) {

                case 1:
                    System.out.print("Ingrese monto de la venta: ");
                    double venta = sc.nextDouble();

                    if (venta <= 0) //  ERROR 2: Punto y coma rompe el if (el bloque se ejecuta siempre)
                    {   //  ERROR 3: Validación incorrecta, debe ser <= 0
                                    
                        System.out.println("Venta invalida");
                    }
                    else {
                        totalVentas += venta;   //  ERROR 4: No acumula, reemplaza el total
                        contadorVentas++;
                        System.out.println("Venta registrada");
                    }
                    break;

                case 2:
                    System.out.println("Total vendido: " + totalVentas);

                    for (int i = 0; i < contadorVentas; i++) {   //  ERROR 5: Se ejecuta una vez de más (debe ser <)
                        System.out.println("Procesando venta...");
                    }
                    break;

                case 3:
                    System.out.println("Saliendo del sistema...");
                    break;

                default:
                    System.out.println("Opcion invalida");
            }
        }

        sc.close();
    }
}
