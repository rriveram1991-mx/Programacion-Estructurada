    public static void main(String[] args) {

        System.out.print("Ingreso familiar: ");

            if (ingreso < 10000) {
                System.out.println("Beca del 100%");

            }

                System.out.println("Beca del 25%");
            }
        }
    }

import java.util.Scanner;

public class Triangulo {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int b = sc.nextInt();
        int c = sc.nextInt();

        if (a + b > c) {
            if (a + c > b) {
                if (b + c > a) {
                    if (a == b && b == c) {
                        System.out.println("Equilatero");
                    } else if (a == b || b == c || a == c) {
                        System.out.println("Isosceles");
                    } else {
                        System.out.println("Escaleno");
                    }
                }
            }
        } else {
            System.out.println("No es triangulo");
        }
    }
}

import java.util.Scanner;

public class SistemaRoles {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Rol (1-Admin, 2-Soporte, 3-Usuario): ");
        int rol = sc.nextInt();

        System.out.print("Nivel (1-3): ");
        int nivel = sc.nextInt();

        switch (rol) {

            case 1:
                switch (nivel) {
                    case 1:
                        System.out.println("Admin básico");
                    case 2:
                        System.out.println("Admin intermedio");
                    case 3:
                        System.out.println("Admin avanzado");
                        break;
                }
                break;

            case 2:
                System.out.println("Soporte técnico");
                break;

            case 3:
                System.out.println("Usuario estándar");
                break;

            default:
                System.out.println("Rol inválido");
        }
    }
}

import java.util.Scanner;

public class CalculadoraAvanzada {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        double a = sc.nextDouble();
        double b = sc.nextDouble();

        System.out.print("Operación (+, -, *, /): ");
        char op = sc.next().charAt(0);

        switch (op) {

            case '+':
                System.out.println("Resultado: " + a + b);
                break;

            case '-':
                System.out.println("Resultado: " + (a - b));
                break;

            case '*':
                System.out.println("Resultado: " + (a * b));
                break;

            case '/':
                System.out.println("Resultado: " + (a / b));
                break;

            default:
                System.out.println("Operación inválida");
        }
    }
}
